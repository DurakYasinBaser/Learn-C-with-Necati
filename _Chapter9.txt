int sum_square(int x, int y)
{
	return x * x + y * y;

int ndigit(int x)
{
	if (x == 0)
		return 1;
	int digit_count = 0;
	while (x) {
		digit_count++;
		x /= 10;
	}
	return digit_count;
}
int main(void)
{
	int number;
	printf("Bir Sayi Giriniz:\n");
	(void)scanf("%d", &number);
	printf("Sayi %d Basamaklidir.\n", ndigit(number));
}

																		scanf() Function
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------

scanf fonksiyonunun bildirimi <stdio.h> baþlýk dosyasýndadýr.
scanf fonksiyonu da printf fonksiyonu gibi variadic bir fonksiyondur.

int scanf(const char*, ...);

scanf fonksiyonu standard input stream'den alýnan karakterleri kullanarak, bizim istediðimiz formatlama özelliklerine göre bizim deðiþken yada deðiþkenlerimize deðer yükleyen program.

buffer(tampon): Bir takým bilgileri depolamak için kullanýlan bir bellek alanýdýr.

scanf fonksiyonunun printf fonksiyonundan farklý olarak call by value deðil, call by referance olmasý mecburidir. 

	NOT:	scanf gunc. conversion specififier'larýný tam olarak öðrenene kadar conversion specifier dýþýnda boþluk karakteri bile vermeyin. Hepsinin ayný birer anlamý var.
	NOT:	scanf function's ýn baþarýsýz olma ihtimali printf'e göre çok daha yüksektir. 
	NOT:	Scanf'in geri dönüþ deðeri set edilen deðiþken sayýsýdýr.

// Bir tam sayinin basamaklarýnýn toplamýný hesaplayan fonksiyonu yazýnýz.

int sum_digit(int x)
{
	int result = 0;

	while (x) {
		result += x % 10;
		x /= 10;

	}

	return result;
}

{

	(void)scanf("%d", &x);
